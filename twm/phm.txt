# # Cell 3: Multi-Domain Feature Extraction Engine
# Extracts time-domain, frequency-domain, time-frequency, and physics-informed features.

# %%
class MultiDomainFeatureExtractor:
    """
    Comprehensive multi-domain feature extraction for machining signals.
    Extracts 12 features per sensor channel = 84 total features per cut.

    Domains:
        - Time domain: statistical + nonlinear features
        - Frequency domain: spectral features
        - Time-frequency domain: wavelet energy features
        - Physics-informed: cutting mechanics features
    """

    def __init__(self, fs=SAMPLING_RATE):
        self.fs = fs

    def time_domain_features(self, x):
        """Extract time-domain statistical features."""
        rms = np.sqrt(np.mean(x**2))
        peak = np.max(np.abs(x))
        crest = peak / (rms + 1e-10)
        kurtosis = float(pd.Series(x).kurtosis())
        skewness = float(pd.Series(x).skew())
        std = np.std(x)
        mean_abs = np.mean(np.abs(x))
        shape_factor = rms / (mean_abs + 1e-10)
        return [rms, peak, crest, kurtosis, skewness, std, mean_abs, shape_factor]

    def frequency_domain_features(self, x):
        """Extract frequency-domain spectral features."""
        n = len(x)
        fft_vals = np.abs(np.fft.rfft(x))
        freqs = np.fft.rfftfreq(n, d=1.0/self.fs)

        # Spectral centroid
        total_power = np.sum(fft_vals**2) + 1e-10
        spectral_centroid = np.sum(freqs * fft_vals**2) / total_power

        # Spectral bandwidth
        spectral_bw = np.sqrt(np.sum((freqs - spectral_centroid)**2 * fft_vals**2) / total_power)

        return [spectral_centroid, spectral_bw]

    def wavelet_energy_features(self, x, wavelet='db4', level=4):
        """Extract wavelet packet energy distribution."""
        coeffs = pywt.wavedec(x, wavelet, level=level)
        energies = [np.sum(c**2) for c in coeffs]
        total = sum(energies) + 1e-10
        # Return normalized energy of detail coefficients at 2 highest levels
        return [energies[1]/total, energies[2]/total]

    def extract_single_channel(self, x):
        """Extract all features for a single sensor channel."""
        tf = self.time_domain_features(x)        # 8 features
        ff = self.frequency_domain_features(x)    # 2 features
        wf = self.wavelet_energy_features(x)      # 2 features
        return tf + ff + wf                        # 12 features total

    def extract_cut_features(self, cut_df):
        """Extract features for all channels of a single cut."""
        features = []
        for col in SENSOR_COLS:
            x = cut_df[col].values
            features.extend(self.extract_single_channel(x))
        return np.array(features, dtype=np.float32)

    def extract_cutter_features(self, cutter_id, n_cuts=NUM_CUTS):
        """Extract features for all cuts of a cutter."""
        print(f"  Extracting features for {cutter_id}...", end=" ")
        all_features = []
        for i in range(1, n_cuts + 1):
            cut_df = load_cut_data(cutter_id, i)
            feat = self.extract_cut_features(cut_df)
            all_features.append(feat)
            if i % 50 == 0:
                print(f"{i}", end=" ")
        print("Done!")
        return np.array(all_features)  # (N_cuts, FEATURE_DIM)

class ToolWearSequenceDataset(Dataset):
    """
    Creates overlapping sequences for temporal modeling.
    Each sample: (feature_seq, wear_seq, rul_seq, flute_wear_seq, time_indices)
    """

    def __init__(self, features, wear, rul, flute_wear, seq_len=SEQ_LEN):
        self.seq_len = seq_len
        self.samples = []

        n = len(features)
        for i in range(n - seq_len):
            feat_seq = features[i:i+seq_len]
            wear_seq = wear[i:i+seq_len]
            rul_seq = rul[i:i+seq_len]
            flute_seq = flute_wear[i:i+seq_len]
            time_idx = np.arange(i, i+seq_len, dtype=np.float32) / n  # Normalized time

            self.samples.append((
                torch.FloatTensor(feat_seq),
                torch.FloatTensor(wear_seq),
                torch.FloatTensor(rul_seq),
                torch.FloatTensor(flute_seq),
                torch.FloatTensor(time_idx),
            ))

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx):
        return self.samples[idx]

# # Cell 6: Physics-Informed Neural ODE Components
# Core architecture: Neural ODE with physics-constrained wear dynamics

# %%
# ========================
# 6A: ODE Function Network
# ========================
class WearRateODE(nn.Module):
    """
    Neural ODE function: dW/dt = f_physics(W, t) + f_neural(z, W, t)

    The wear rate is decomposed into:
    - Physics component: Monotonic increasing rate based on Archard/Usui principles
    - Neural residual: Learned correction capturing unknown tribological effects

    Key constraint: dW/dt >= 0 (wear is irreversible)
    """

    def __init__(self, latent_dim, hidden_dim=128):
        super().__init__()
        self.latent_dim = latent_dim

        # Physics-inspired wear rate (parameterized Archard-like model)
        # dW/dt_physics = a * exp(b * W) * (1 + c * t)
        self.physics_a = nn.Parameter(torch.tensor(0.1))
        self.physics_b = nn.Parameter(torch.tensor(0.01))
        self.physics_c = nn.Parameter(torch.tensor(0.1))

        # Neural residual network
        self.neural_net = nn.Sequential(
            nn.Linear(latent_dim + 2, hidden_dim),  # +2 for W and t
            nn.SiLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.SiLU(),
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.SiLU(),
            nn.Linear(hidden_dim // 2, 1),
        )

        # Softplus to ensure non-negative wear rate
        self.softplus = nn.Softplus(beta=5)

    def physics_rate(self, W, t):
        """Archard-Usui inspired physics wear rate."""
        a = self.softplus(self.physics_a)
        b = self.softplus(self.physics_b)
        c = self.softplus(self.physics_c)
        rate = a * torch.exp(b * W) * (1.0 + c * t)
        return rate

    def forward(self, t_val, state):
        """
        Args:
            t_val: scalar time (normalized)
            state: (batch, latent_dim + 1) where last dim is W
        Returns:
            d_state/dt: (batch, latent_dim + 1)
        """
        z = state[:, :-1]        # Latent state: (batch, latent_dim)
        W = state[:, -1:]        # Wear state: (batch, 1)

        t_input = t_val * torch.ones_like(W)

        # Physics rate
        phys_rate = self.physics_rate(W, t_input)

        # Neural residual
        neural_input = torch.cat([z, W, t_input], dim=-1)
        neural_rate = self.neural_net(neural_input)

        # Total wear rate (guaranteed non-negative)
        total_rate = self.softplus(phys_rate + neural_rate)

        # Latent dynamics (unconstrained)
        dz = torch.zeros_like(z)  # Latent state evolves through encoder, not ODE

        dW = total_rate
        d_state = torch.cat([dz, dW], dim=-1)
        return d_state


# ========================
# 6B: Simple Euler ODE Solver (Memory efficient for GTX 1080)
# ========================
class EulerODESolver(nn.Module):
    """
    Fixed-step Euler solver for Neural ODE.
    Memory efficient alternative to adaptive solvers.
    """

    def __init__(self, ode_func, n_steps=10):
        super().__init__()
        self.ode_func = ode_func
        self.n_steps = n_steps

    def forward(self, initial_state, t_span):
        """
        Solve ODE from t_span[0] to t_span[1].

        Args:
            initial_state: (batch, state_dim)
            t_span: (2,) tensor [t0, t1]
        Returns:
            final_state: (batch, state_dim)
        """
        t0, t1 = t_span[0], t_span[1]
        dt = (t1 - t0) / self.n_steps
        state = initial_state

        for i in range(self.n_steps):
            t_val = t0 + i * dt
            d_state = self.ode_func(t_val, state)
            state = state + dt * d_state

        return state

# # Cell 7: Complete PINN Framework Architecture

# %%
class PhysicsInformedNeuralWearModel(nn.Module):
    """
    ===================================================================
    PINN-NDE: Physics-Informed Neural Differential Equation Framework
    for Tool Wear Degradation Modeling
    ===================================================================

    Architecture:
    1. Feature Encoder: Maps raw features to latent representation
    2. Temporal Encoder: GRU-based sequence modeling
    3. Neural ODE: Physics-constrained wear evolution
    4. Multi-Head Decoder: Wear, RUL, and flute-level predictions
    5. Uncertainty Head: Aleatoric + epistemic uncertainty

    Physics Constraints (enforced in loss):
    - Monotonicity: dW/dt >= 0 (irreversible degradation)
    - Energy consistency: Work ∝ material removal + friction
    - Archard proportionality: W ∝ F·L/H
    - Phase-aware dynamics: Different ODE branches for wear phases
    """

    def __init__(self, input_dim=FEATURE_DIM, hidden_dim=HIDDEN_DIM,
                 latent_dim=LATENT_DIM, n_flutes=NUM_FLUTES, ode_steps=10):
        super().__init__()

        self.latent_dim = latent_dim
        self.hidden_dim = hidden_dim

        # ===== 1. Feature Encoder =====
        self.feature_encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
            nn.SiLU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
            nn.SiLU(),
            nn.Linear(hidden_dim, latent_dim),
        )

        # ===== 2. Temporal Encoder (Bidirectional GRU) =====
        self.temporal_encoder = nn.GRU(
            input_size=latent_dim,
            hidden_size=hidden_dim // 2,
            num_layers=2,
            batch_first=True,
            bidirectional=True,
            dropout=0.1,
        )
        self.temporal_proj = nn.Linear(hidden_dim, latent_dim)

        # ===== 3. Neural ODE =====
        self.ode_func = WearRateODE(latent_dim, hidden_dim)
        self.ode_solver = EulerODESolver(self.ode_func, n_steps=ode_steps)

        # Initial wear state predictor
        self.initial_wear_net = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim // 2),
            nn.SiLU(),
            nn.Linear(hidden_dim // 2, 1),
            nn.Sigmoid(),  # Normalized wear [0, 1]
        )

        # ===== 4. Multi-Head Decoder =====
        # Total wear decoder
        self.wear_decoder = nn.Sequential(
            nn.Linear(latent_dim + 1, hidden_dim // 2),  # +1 for ODE wear state
            nn.SiLU(),
            nn.Linear(hidden_dim // 2, 1),
            nn.Sigmoid(),
        )

        # Flute-level wear decoder
        self.flute_decoder = nn.Sequential(
            nn.Linear(latent_dim + 1, hidden_dim // 2),
            nn.SiLU(),
            nn.Linear(hidden_dim // 2, n_flutes),
            nn.Sigmoid(),
        )

        # RUL decoder
        self.rul_decoder = nn.Sequential(
            nn.Linear(latent_dim + 1, hidden_dim // 2),
            nn.SiLU(),
            nn.Linear(hidden_dim // 2, 1),
            nn.Sigmoid(),
        )

        # ===== 5. Uncertainty Head =====
        self.uncertainty_head = nn.Sequential(
            nn.Linear(latent_dim + 1, hidden_dim // 4),
            nn.SiLU(),
            nn.Linear(hidden_dim // 4, 2),  # log_var for wear and RUL
        )

        # ===== 6. Phase Classifier =====
        # Classifies wear phase: 0=initial, 1=steady, 2=accelerated
        self.phase_classifier = nn.Sequential(
            nn.Linear(latent_dim + 1, hidden_dim // 4),
            nn.SiLU(),
            nn.Linear(hidden_dim // 4, 3),
        )

        # ===== 7. Latent Physics Inference Head =====
        # Infers latent temperature and stress from sensor features
        self.latent_physics = nn.Sequential(
            nn.Linear(latent_dim, hidden_dim // 4),
            nn.SiLU(),
            nn.Linear(hidden_dim // 4, 2),  # [temperature_proxy, stress_proxy]
            nn.Softplus(),  # Non-negative physical quantities
        )

    def forward(self, feat_seq, time_seq):
        """
        Forward pass through the complete framework.

        Args:
            feat_seq: (batch, seq_len, input_dim) - Feature sequences
            time_seq: (batch, seq_len) - Normalized time indices

        Returns:
            dict with all predictions and intermediate states
        """
        batch_size, seq_len, _ = feat_seq.shape

        # 1. Encode features
        encoded = self.feature_encoder(feat_seq)  # (B, T, latent_dim)

        # 2. Temporal encoding
        temporal_out, _ = self.temporal_encoder(encoded)  # (B, T, hidden_dim)
        z = self.temporal_proj(temporal_out)  # (B, T, latent_dim)

        # 3. Neural ODE for wear evolution
        # Get initial wear state from first timestep
        W0 = self.initial_wear_net(z[:, 0, :])  # (B, 1)

        # Solve ODE step by step through the sequence
        wear_states = [W0]
        physics_rates = []
        current_W = W0

        for t in range(1, seq_len):
            # Build state: [latent, wear]
            state = torch.cat([z[:, t-1, :], current_W], dim=-1)

            t_start = time_seq[:, t-1].mean()
            t_end = time_seq[:, t].mean()
            t_span = torch.tensor([t_start, t_end], device=feat_seq.device)

            # Solve ODE
            next_state = self.ode_solver(state, t_span)
            current_W = next_state[:, -1:]

            wear_states.append(current_W)

            # Track physics rate for analysis
            with torch.no_grad():
                pr = self.ode_func.physics_rate(current_W, t_end * torch.ones_like(current_W))
                physics_rates.append(pr)

        # Stack wear states: (B, T, 1)
        wear_ode = torch.stack(wear_states, dim=1)

        # 4. Decode predictions at each timestep
        decoder_input = torch.cat([z, wear_ode], dim=-1)  # (B, T, latent+1)

        wear_pred = self.wear_decoder(decoder_input).squeeze(-1)     # (B, T)
        flute_pred = self.flute_decoder(decoder_input)               # (B, T, 3)
        rul_pred = self.rul_decoder(decoder_input).squeeze(-1)       # (B, T)

        # 5. Uncertainty estimation
        uncertainty = self.uncertainty_head(decoder_input)  # (B, T, 2)
        log_var_wear = uncertainty[:, :, 0]
        log_var_rul = uncertainty[:, :, 1]

        # 6. Phase classification
        phase_logits = self.phase_classifier(decoder_input)  # (B, T, 3)

        # 7. Latent physics inference
        latent_phys = self.latent_physics(z)  # (B, T, 2)

        return {
            'wear_pred': wear_pred,
            'flute_pred': flute_pred,
            'rul_pred': rul_pred,
            'wear_ode': wear_ode.squeeze(-1),
            'log_var_wear': log_var_wear,
            'log_var_rul': log_var_rul,
            'phase_logits': phase_logits,
            'latent_temperature': latent_phys[:, :, 0],
            'latent_stress': latent_phys[:, :, 1],
            'latent_z': z,
            'physics_rates': physics_rates,
        }

# # Cell 8: Physics-Informed Loss Function

# %%
class PhysicsInformedLoss(nn.Module):
    """
    Multi-objective physics-informed loss function.

    Components:
    1. Data fidelity: MSE for wear, RUL, and flute-level predictions
    2. Monotonicity constraint: Penalize non-monotonic wear trajectories
    3. Irreversibility constraint: dW/dt >= 0
    4. Energy consistency: Force-wear proportionality (Archard)
    5. Uncertainty regularization: NLL with learned variance
    6. Smoothness regularization: Penalize abrupt wear rate changes
    7. ODE consistency: Decoded wear should match ODE-predicted wear
    """

    def __init__(self, lambda_mono=5.0, lambda_irreversible=10.0,
                 lambda_energy=1.0, lambda_smooth=0.5,
                 lambda_ode=2.0, lambda_flute=1.0,
                 lambda_rul=1.0, lambda_uncertainty=0.1):
        super().__init__()

        self.lambda_mono = lambda_mono
        self.lambda_irreversible = lambda_irreversible
        self.lambda_energy = lambda_energy
        self.lambda_smooth = lambda_smooth
        self.lambda_ode = lambda_ode
        self.lambda_flute = lambda_flute
        self.lambda_rul = lambda_rul
        self.lambda_uncertainty = lambda_uncertainty

        self.mse = nn.MSELoss()
        self.huber = nn.SmoothL1Loss()

    def monotonicity_loss(self, wear_pred):
        """Penalize violations of monotonic increase in wear."""
        diff = wear_pred[:, 1:] - wear_pred[:, :-1]
        violations = F.relu(-diff)  # Only penalize decreases
        return torch.mean(violations ** 2)

    def irreversibility_loss(self, wear_ode):
        """Strongly penalize any decrease in ODE wear state."""
        diff = wear_ode[:, 1:] - wear_ode[:, :-1]
        violations = F.relu(-diff)
        return torch.mean(violations ** 2)

    def smoothness_loss(self, wear_pred):
        """Penalize abrupt changes in wear rate (2nd derivative)."""
        first_diff = wear_pred[:, 1:] - wear_pred[:, :-1]
        second_diff = first_diff[:, 1:] - first_diff[:, :-1]
        return torch.mean(second_diff ** 2)

    def ode_consistency_loss(self, wear_pred, wear_ode):
        """Decoded wear should be consistent with ODE-predicted wear."""
        return self.mse(wear_pred, wear_ode)

    def nll_loss(self, pred, target, log_var):
        """Negative log-likelihood with learned variance."""
        precision = torch.exp(-log_var)
        nll = 0.5 * (precision * (pred - target)**2 + log_var)
        return torch.mean(nll)

    def forward(self, outputs, targets):
        """
        Compute total physics-informed loss.

        Args:
            outputs: dict from model forward pass
            targets: dict with 'wear', 'rul', 'flute_wear' tensors
        """
        losses = {}

        # 1. Data fidelity losses
        losses['wear'] = self.huber(outputs['wear_pred'], targets['wear'])
        losses['rul'] = self.huber(outputs['rul_pred'], targets['rul'])
        losses['flute'] = self.huber(outputs['flute_pred'], targets['flute_wear'])

        # 2. Physics constraint losses
        losses['monotonicity'] = self.monotonicity_loss(outputs['wear_pred'])
        losses['irreversibility'] = self.irreversibility_loss(outputs['wear_ode'])
        losses['smoothness'] = self.smoothness_loss(outputs['wear_pred'])

        # 3. ODE consistency
        losses['ode_consistency'] = self.ode_consistency_loss(
            outputs['wear_pred'], outputs['wear_ode'])

        # 4. Uncertainty-aware loss
        losses['nll_wear'] = self.nll_loss(
            outputs['wear_pred'], targets['wear'], outputs['log_var_wear'])
        losses['nll_rul'] = self.nll_loss(
            outputs['rul_pred'], targets['rul'], outputs['log_var_rul'])

        # Total loss
        total = (
            losses['wear']
            + self.lambda_rul * losses['rul']
            + self.lambda_flute * losses['flute']
            + self.lambda_mono * losses['monotonicity']
            + self.lambda_irreversible * losses['irreversibility']
            + self.lambda_smooth * losses['smoothness']
            + self.lambda_ode * losses['ode_consistency']
            + self.lambda_uncertainty * (losses['nll_wear'] + losses['nll_rul'])
        )

        losses['total'] = total
        return losses


criterion = PhysicsInformedLoss()
print("Physics-Informed Loss Function initialized with components:")

# # Cell 9: Training Loop with Physics-Aware Scheduling

# %%
def train_model(model, train_loader, criterion, num_epochs=NUM_EPOCHS,
                lr=LEARNING_RATE, device=DEVICE):
    """
    Training loop with:
    - Cosine annealing warm restarts
    - Gradient clipping
    - Physics constraint curriculum (gradually increase physics penalty)
    - Best model checkpointing
    """

    optimizer = torch.optim.AdamW(model.parameters(), lr=lr,
                                   weight_decay=WEIGHT_DECAY)
    scheduler = CosineAnnealingWarmRestarts(optimizer, T_0=50, T_mult=2, eta_min=1e-6)

    history = {
        'total': [], 'wear': [], 'rul': [], 'monotonicity': [],
        'irreversibility': [], 'ode_consistency': [], 'lr': [],
    }

    best_loss = float('inf')
    best_state = None
    patience_counter = 0
    patience = 40

    print(f"\n{'='*60}")
    print(f"TRAINING STARTED")
    print(f"{'='*60}")
    print(f"Epochs: {num_epochs}, LR: {lr}, Batch: {BATCH_SIZE}")
    print(f"{'='*60}\n")

    for epoch in range(1, num_epochs + 1):
        model.train()
        epoch_losses = {k: 0.0 for k in history.keys() if k != 'lr'}
        n_batches = 0

        # Physics constraint curriculum: ramp up over first 50 epochs
        curriculum_weight = min(1.0, epoch / 50.0)
        criterion.lambda_mono = 5.0 * curriculum_weight
        criterion.lambda_irreversible = 10.0 * curriculum_weight

        for batch in train_loader:
            feat_seq, wear_seq, rul_seq, flute_seq, time_seq = batch
            feat_seq = feat_seq.to(device)
            wear_seq = wear_seq.to(device)
            rul_seq = rul_seq.to(device)
            flute_seq = flute_seq.to(device)
            time_seq = time_seq.to(device)

            # Forward pass
            outputs = model(feat_seq, time_seq)

            targets = {
                'wear': wear_seq,
                'rul': rul_seq,
                'flute_wear': flute_seq,
            }

            losses = criterion(outputs, targets)

            # Backward pass
            optimizer.zero_grad()
            losses['total'].backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
            optimizer.step()

            for k in epoch_losses:
                if k in losses:
                    epoch_losses[k] += losses[k].item()
            n_batches += 1

        scheduler.step()

        # Average losses
        for k in epoch_losses:
            epoch_losses[k] /= max(n_batches, 1)
            history[k].append(epoch_losses[k])
        history['lr'].append(optimizer.param_groups[0]['lr'])

        # Best model tracking
        if epoch_losses['total'] < best_loss:
            best_loss = epoch_losses['total']
            best_state = {k: v.cpu().clone() for k, v in model.state_dict().items()}
            patience_counter = 0
        else:
            patience_counter += 1

        # Print progress
        if epoch % 10 == 0 or epoch == 1:
            print(f"Epoch {epoch:03d}/{num_epochs} | "
                  f"Total: {epoch_losses['total']:.6f} | "
                  f"Wear: {epoch_losses['wear']:.6f} | "
                  f"RUL: {epoch_losses['rul']:.6f} | "
                  f"Mono: {epoch_losses['monotonicity']:.6f} | "
                  f"ODE: {epoch_losses['ode_consistency']:.6f} | "
                  f"LR: {optimizer.param_groups[0]['lr']:.2e}")

        # Early stopping
        if patience_counter >= patience:
            print(f"\nEarly stopping at epoch {epoch} (patience={patience})")
            break

    # Restore best model
    if best_state is not None:
        model.load_state_dict(best_state)
        print(f"\nRestored best model (loss: {best_loss:.6f})")

    return history


# ===== TRAIN =====
history = train_model(model, train_loader, criterion)

@torch.no_grad()
def predict_full_trajectory(model, features, seq_len=SEQ_LEN, device=DEVICE):
    """
    Run inference on full cutter trajectory using sliding window.
    Returns predictions for every cut by averaging overlapping windows.

    Args:
        features: (N_cuts, feature_dim) normalized features
    Returns:
        dict with wear, rul, flute, uncertainty predictions for each cut
    """
    model.eval()
    n_cuts = len(features)

    # Accumulators for averaging overlapping predictions
    wear_accum = np.zeros(n_cuts)
    rul_accum = np.zeros(n_cuts)
    flute_accum = np.zeros((n_cuts, NUM_FLUTES))
    var_wear_accum = np.zeros(n_cuts)
    var_rul_accum = np.zeros(n_cuts)
    phase_accum = np.zeros((n_cuts, 3))
    temp_accum = np.zeros(n_cuts)
    stress_accum = np.zeros(n_cuts)
    count = np.zeros(n_cuts)

    for i in range(n_cuts - seq_len):
        feat_seq = torch.FloatTensor(features[i:i+seq_len]).unsqueeze(0).to(device)
        time_seq = torch.FloatTensor(
            np.arange(i, i+seq_len, dtype=np.float32) / n_cuts
        ).unsqueeze(0).to(device)

        outputs = model(feat_seq, time_seq)

        wear = outputs['wear_pred'][0].cpu().numpy()
        rul = outputs['rul_pred'][0].cpu().numpy()
        flute = outputs['flute_pred'][0].cpu().numpy()
        var_w = torch.exp(outputs['log_var_wear'][0]).cpu().numpy()
        var_r = torch.exp(outputs['log_var_rul'][0]).cpu().numpy()
        phase = F.softmax(outputs['phase_logits'][0], dim=-1).cpu().numpy()
        temp = outputs['latent_temperature'][0].cpu().numpy()
        stress = outputs['latent_stress'][0].cpu().numpy()

        for j in range(seq_len):
            idx = i + j
            wear_accum[idx] += wear[j]
            rul_accum[idx] += rul[j]
            flute_accum[idx] += flute[j]
            var_wear_accum[idx] += var_w[j]
            var_rul_accum[idx] += var_r[j]
            phase_accum[idx] += phase[j]
            temp_accum[idx] += temp[j]
            stress_accum[idx] += stress[j]
            count[idx] += 1

    # Avoid division by zero
    count = np.maximum(count, 1)

    results = {
        'wear': wear_accum / count,
        'rul': rul_accum / count,
        'flute': flute_accum / count[:, None],
        'wear_std': np.sqrt(var_wear_accum / count),
        'rul_std': np.sqrt(var_rul_accum / count),
        'phase_probs': phase_accum / count[:, None],
        'latent_temperature': temp_accum / count,
        'latent_stress': stress_accum / count,
    }

    return results

# # Cell 12: Quantitative Evaluation Metrics

# %%
def compute_metrics(y_true, y_pred):
    """Compute comprehensive regression metrics."""
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    mae = mean_absolute_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / (y_true + 1e-10))) * 100

    # Correlation metrics
    pearson_r, _ = pearsonr(y_true, y_pred)
    spearman_r, _ = spearmanr(y_true, y_pred)

    # Monotonicity score
    diffs = np.diff(y_pred)
    mono_score = np.sum(diffs >= 0) / len(diffs) * 100

    return {
        'RMSE': rmse, 'MAE': mae, 'R²': r2, 'MAPE': mape,
        'Pearson': pearson_r, 'Spearman': spearman_r,
        'Monotonicity%': mono_score
    }

def train_ablation_model(model_class, train_loader, config_name,
                         lambda_mono=0.0, lambda_irreversible=0.0,
                         lambda_ode=0.0, num_epochs=150, device=DEVICE):
    """Train a model variant for ablation study."""
    print(f"\n--- Ablation: {config_name} ---")

    ablation_model = model_class().to(device)
    ablation_criterion = PhysicsInformedLoss(
        lambda_mono=lambda_mono,
        lambda_irreversible=lambda_irreversible,
        lambda_ode=lambda_ode,
    )

    optimizer = torch.optim.AdamW(ablation_model.parameters(), lr=LEARNING_RATE,
                                   weight_decay=WEIGHT_DECAY)
    scheduler = CosineAnnealingWarmRestarts(optimizer, T_0=50, T_mult=2, eta_min=1e-6)

    best_loss = float('inf')
    best_state = None

    for epoch in range(1, num_epochs + 1):
        ablation_model.train()
        total_loss = 0
        n_batches = 0

        for batch in train_loader:
            feat_seq, wear_seq, rul_seq, flute_seq, time_seq = batch
            feat_seq = feat_seq.to(device)
            wear_seq = wear_seq.to(device)
            rul_seq = rul_seq.to(device)
            flute_seq = flute_seq.to(device)
            time_seq = time_seq.to(device)

            outputs = ablation_model(feat_seq, time_seq)
            targets = {'wear': wear_seq, 'rul': rul_seq, 'flute_wear': flute_seq}
            losses = ablation_criterion(outputs, targets)

            optimizer.zero_grad()
            losses['total'].backward()
            torch.nn.utils.clip_grad_norm_(ablation_model.parameters(), max_norm=1.0)
            optimizer.step()

            total_loss += losses['total'].item()
            n_batches += 1

        scheduler.step()
        avg_loss = total_loss / max(n_batches, 1)

        if avg_loss < best_loss:
            best_loss = avg_loss
            best_state = {k: v.cpu().clone() for k, v in ablation_model.state_dict().items()}

        if epoch % 50 == 0:
            print(f"  Epoch {epoch}: Loss = {avg_loss:.6f}")

    if best_state:
        ablation_model.load_state_dict(best_state)

    return ablation_model

fig, axes = plt.subplots(1, 3, figsize=(18, 5))

for idx, cid in enumerate(TRAIN_CUTTERS):
    ax = axes[idx]
    y_true = train_wear_normalized[cid]

    min_len = min(len(y_true), len(y_pred))

    errors = np.abs(y_true[:min_len] - y_pred[:min_len])
    stds = y_std[:min_len]

    # Check calibration: what % of errors fall within k*sigma
    k_values = np.linspace(0.1, 3.0, 30)
    coverage = []
    expected = []
    for k in k_values:
        cov = np.mean(errors <= k * stds) * 100
        exp = (2 * (1 - 0.5 * (1 + np.math.erf(-k/np.sqrt(2))))) * 100  # Rough approx
        # More precise: use scipy
        from scipy.stats import norm
        exp_precise = (norm.cdf(k) - norm.cdf(-k)) * 100
        coverage.append(cov)
        expected.append(exp_precise)

fig, axes = plt.subplots(1, 3, figsize=(18, 5))

for idx, cid in enumerate(TRAIN_CUTTERS):
    ax = axes[idx]

    model.eval()
    physics_contrib = []
    total_rates = []

    with torch.no_grad():
        for i in range(NUM_CUTS):
            W = torch.tensor([[train_predictions[cid]['wear'][i]]],
                           dtype=torch.float32).to(DEVICE)
            t = torch.tensor([[i / NUM_CUTS]], dtype=torch.float32).to(DEVICE)

            phys_rate = model.ode_func.physics_rate(W, t)
            physics_contrib.append(phys_rate.item())

    physics_arr = np.array(physics_contrib)
    cuts = np.arange(1, NUM_CUTS + 1)

    # Estimate total rate from wear differences
    wear_pred = train_predictions[cid]['wear']
    total_rate_est = np.gradient(wear_pred)
    total_rate_est = np.maximum(total_rate_est, 0)

    neural_contrib = total_rate_est - physics_arr[:len(total_rate_est)]

    ax.fill_between(cuts[:len(physics_arr)], 0, physics_arr,
                    alpha=0.4, color='#2196F3', label='Physics Component')
    ax.fill_between(cuts[:len(neural_contrib)], physics_arr[:len(neural_contrib)],
                    physics_arr[:len(neural_contrib)] + np.maximum(neural_contrib, 0),
                    alpha=0.4, color='#F44336', label='Neural Residual')
    ax.plot(cuts[:len(total_rate_est)], total_rate_est,
            'k-', linewidth=1.5, label='Total Rate')

    ax.set_title(f'{cid.upper()} - Rate Decomposition', fontweight='bold')
    ax.set_xlabel('Cut Number')
    ax.set_ylabel('Wear Rate dW/dt')
    ax.legend(fontsize=9)
    ax.grid(True, alpha=0.3)

plt.suptitle('PINN-NDE: Physics vs Neural Wear Rate Decomposition',
             fontsize=14, fontweight='bold', y=1.03)
